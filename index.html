⚙️ เชื่อมต่อ Google Sheets
วิธีเชื่อมต่อ Google Sheets:

1. สร้าง Google Sheet ใหม่ ตั้งชื่อ sheet แรกว่า Memos
2. Extensions → Apps Script → วางโค้ดนี้:

const SHEET_NAME = 'Memos'; function doGet(e) { if (e.parameter.action === 'getAll') { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet || sheet.getLastRow() <= 1) return output({ status:'ok', data:[] }); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const data = rows.slice(1).map(r => { const obj = {}; headers.forEach((h, i) => { let val = r[i]; if (h === 'date') { // force YYYY-MM-DD string if (val instanceof Date) { const y = val.getFullYear(); const m = String(val.getMonth()+1).padStart(2,'0'); const d = String(val.getDate()).padStart(2,'0'); val = y+'-'+m+'-'+d; } else { val = String(val).replace(/^'+/, '').trim(); } } if (h === 'id') val = String(val); obj[h] = val; }); return obj; }).reverse(); return output({ status:'ok', data }); } return output({ status:'error' }); } function doPost(e) { const body = JSON.parse(e.postData.contents); if (body.action === 'add') { const r = body.record; const sheet = getOrCreateSheet(); if (sheet.getLastRow() === 0) sheet.appendRow(['id','memoNum','name','position', 'dept','subject','date','desc','timestamp']); // prefix ' เพื่อกัน Sheets แปลง date เป็น Date object sheet.appendRow([String(r.id), r.memoNum, r.name, r.position, r.dept, r.subject, "'" + r.date, r.desc||'', r.timestamp]); return output({ status:'ok' }); } return output({ status:'error' }); } function getOrCreateSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); return ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME); } function output(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
3. Deploy → New deployment → Web app
   Execute as: Me | Access: Anyone
4. คัดลอก URL → วางใน GAS_URL ในโค้ด

⚠️ หมายเหตุ: ถ้าแก้โค้ด GAS ต้อง Deploy ใหม่ (New deployment) ทุกครั้ง ไม่ใช่แค่ Save
