# เลขหนังสือ ชื่อ-สกุล ตำแหน่ง ฝ่าย เรื่อง วันที่ จัดการ
📭

ยังไม่มีข้อมูล กรุณาส่งคำขอครั้งแรก

⚙️ เชื่อมต่อ Google Sheets
วิธีเชื่อมต่อ Google Sheets:

1. สร้าง Google Sheet ใหม่
2. ไปที่ Extensions → Apps Script
3. วางโค้ด GAS ด้านล่าง:

function doPost(e){
 var d=JSON.parse(e.postData.contents);
 var ss=SpreadsheetApp.getActiveSpreadsheet();
 var sh=ss.getSheetByName('Memos')||ss.insertSheet('Memos');
 if(sh.getLastRow()===0)sh.appendRow(['ID','เลขหนังสือ','ชื่อ','ตำแหน่ง','ฝ่าย','เรื่อง','วันที่','รายละเอียด','เวลา']);
 sh.appendRow([d.id,d.memoNum,d.name,d.position,d.dept,d.subject,d.date,d.desc,d.timestamp]);
 return ContentService.createTextOutput('OK');
}
4. Deploy → Web App → Anyone can access
5. คัดลอก URL ใส่ใน GAS_URL ในโค้ด
